Starlight is configured in your `astro.config.mjs` file using the `starlight()` integration. This page documents all available configuration options.

## Basic Configuration

<ParamField path="title" type="string | Record<string, string>" required>
  Title for your website. Used in metadata and as the browser tab title.
  
  For multilingual sites, provide an object with titles for each locale:
  
  ```js
  starlight({
    title: {
      en: 'My Docs',
      fr: 'Ma Documentation'
    }
  })
  ```
</ParamField>

<ParamField path="description" type="string">
  Description metadata for your website. Used in page metadata when a page doesn't define its own description.
</ParamField>

<ParamField path="tagline" type="string">
  The tagline for your website. Displayed on the homepage if using the splash template.
</ParamField>

## Logo Configuration

<ParamField path="logo" type="object">
  Set a logo image to show in the navigation bar alongside or instead of the site title.
  
  ### Single Logo
  
  <ParamField path="logo.src" type="string">
    Source path to the image file.
  </ParamField>
  
  <ParamField path="logo.alt" type="string" default="''">
    Alternative text description of the logo for accessibility.
  </ParamField>
  
  <ParamField path="logo.replacesTitle" type="boolean" default="false">
    Set to `true` to hide the site title text and only show the logo.
  </ParamField>
  
  ### Dark/Light Mode Logos
  
  <ParamField path="logo.light" type="string">
    Source path to the image file to use in light mode.
  </ParamField>
  
  <ParamField path="logo.dark" type="string">
    Source path to the image file to use in dark mode.
  </ParamField>
</ParamField>

## Social Links

<ParamField path="social" type="array">
  Optional array of social media accounts for this site. Each link requires:
  
  <ParamField path="social[].icon" type="string" required>
    Icon identifier (e.g., 'github', 'discord', 'mastodon', 'twitter').
  </ParamField>
  
  <ParamField path="social[].label" type="string" required>
    Accessible label for the social link.
  </ParamField>
  
  <ParamField path="social[].href" type="string" required>
    URL of the social media profile or page.
  </ParamField>
  
  ```js
  starlight({
    social: [
      { icon: 'github', label: 'GitHub', href: 'https://github.com/withastro/starlight' },
      { icon: 'discord', label: 'Discord', href: 'https://astro.build/chat' },
    ]
  })
  ```
</ParamField>

## Navigation

<ParamField path="sidebar" type="array">
  Configure your site's sidebar navigation items. Can include links, groups, and autogenerated sections.
  
  ### Link Items
  
  <ParamField path="label" type="string" required>
    The visible label for this item in the sidebar.
  </ParamField>
  
  <ParamField path="link" type="string" required>
    The link to this item's content. Can be a relative link or full URL.
  </ParamField>
  
  <ParamField path="badge" type="string | object">
    Adds a badge to the sidebar item. Can be a string or an object with `text` and `variant` ('note', 'tip', 'caution', 'danger', 'success', 'default').
  </ParamField>
  
  <ParamField path="attrs" type="object">
    HTML attributes to add to the link element.
  </ParamField>
  
  ### Groups
  
  <ParamField path="label" type="string" required>
    The visible label for this group.
  </ParamField>
  
  <ParamField path="items" type="array" required>
    Array of links and subcategories to display in this group.
  </ParamField>
  
  <ParamField path="collapsed" type="boolean" default="false">
    Whether this group should be collapsed by default.
  </ParamField>
  
  ### Autogenerate
  
  <ParamField path="autogenerate" type="object">
    Enable autogenerating a sidebar category from a specific docs directory.
    
    <ParamField path="autogenerate.directory" type="string" required>
      The directory to generate sidebar items for.
    </ParamField>
    
    <ParamField path="autogenerate.collapsed" type="boolean">
      Whether autogenerated subgroups should be collapsed by default.
    </ParamField>
  </ParamField>
</ParamField>

## Localization

<ParamField path="locales" type="object">
  Configure locales for internationalization (i18n). Each key is a locale code (or 'root' for the default locale).
  
  <ParamField path="locales[key].label" type="string" required>
    The label for this language to show in UI (e.g., "English", "简体中文").
  </ParamField>
  
  <ParamField path="locales[key].lang" type="string">
    The BCP-47 tag for this language (e.g., "en", "zh-CN"). Defaults to the key.
  </ParamField>
  
  <ParamField path="locales[key].dir" type="'ltr' | 'rtl'" default="'ltr'">
    The writing direction of this language.
  </ParamField>
  
  ```js
  starlight({
    locales: {
      root: { label: 'English', lang: 'en' },
      fr: { label: 'Français', lang: 'fr' },
      ar: { label: 'العربية', dir: 'rtl' },
    }
  })
  ```
</ParamField>

<ParamField path="defaultLocale" type="string">
  Specify the default language for this site. The default locale provides fallback content where translations are missing.
</ParamField>

## Table of Contents

<ParamField path="tableOfContents" type="boolean | object" default="{ minHeadingLevel: 2, maxHeadingLevel: 3 }">
  Configure the defaults for the table of contents on each page. Can be set to `false` to disable.
  
  <ParamField path="tableOfContents.minHeadingLevel" type="number" default="2">
    The level to start including headings at (1-6).
  </ParamField>
  
  <ParamField path="tableOfContents.maxHeadingLevel" type="number" default="3">
    The level to stop including headings at (1-6).
  </ParamField>
</ParamField>

## Page Footer

<ParamField path="editLink" type="object" default="{}">
  Enable and configure "Edit this page" links.
  
  <ParamField path="editLink.baseUrl" type="string (URL)">
    Set the base URL for edit links. The final link will be `baseUrl` + the current page path.
    
    ```js
    starlight({
      editLink: {
        baseUrl: 'https://github.com/myuser/myrepo/edit/main/'
      }
    })
    ```
  </ParamField>
</ParamField>

<ParamField path="lastUpdated" type="boolean" default="false">
  Define if the last update date should be visible in the page footer.
</ParamField>

<ParamField path="pagination" type="boolean" default="true">
  Define if the previous and next page links should be visible in the page footer.
</ParamField>

<ParamField path="credits" type="boolean" default="false">
  Enable displaying a "Built with Starlight" link in your site's footer.
</ParamField>

## Head and Metadata

<ParamField path="head" type="array">
  Add extra tags to your site's `<head>`. Useful for analytics, external stylesheets, etc.
  
  Each entry requires:
  - `tag`: HTML tag name (e.g., 'script', 'link', 'meta')
  - `attrs`: Object of HTML attributes
  
  ```js
  starlight({
    head: [
      {
        tag: 'script',
        attrs: {
          src: 'https://cdn.usefathom.com/script.js',
          'data-site': 'MY-FATHOM-ID',
          defer: true,
        },
      },
    ],
  })
  ```
</ParamField>

<ParamField path="favicon" type="string">
  The default favicon for your site. Should be a path to an image in the `public/` directory (e.g., '/favicon.svg').
</ParamField>

<ParamField path="titleDelimiter" type="string" default="'|'">
  Will be used as title delimiter in the generated `<title>` tag (e.g., "Page Title | Site Title").
</ParamField>

## Styling

<ParamField path="customCss" type="array" default="[]">
  Provide CSS files to customize the look and feel of your Starlight site.
  
  Supports local CSS files relative to the root of your project (e.g., '/src/custom.css') and CSS installed as npm modules (e.g., '@fontsource/roboto').
  
  ```js
  starlight({
    customCss: [
      './src/custom-styles.css',
      '@fontsource/roboto',
    ]
  })
  ```
  
  **Note:** CSS files must be in the `src/` directory, not `public/`.
</ParamField>

## Search

<ParamField path="pagefind" type="boolean | object" default="true">
  Configure Starlight's default site search provider Pagefind. Set to `false` to disable indexing.
  
  When enabled, accepts Pagefind configuration options. See the [Pagefind documentation](https://pagefind.app/docs/config-options/) for available options.
</ParamField>

## Code Blocks

<ParamField path="expressiveCode" type="boolean | object" default="true">
  Define how code blocks are rendered by passing options to Expressive Code, or disable the integration by passing `false`.
  
  When enabled, accepts Expressive Code configuration options. See the [Expressive Code documentation](https://expressive-code.com/reference/configuration/) for available options.
</ParamField>

## Components

<ParamField path="components" type="object">
  Specify paths to components that should override Starlight's default components.
  
  Keys are component names, values are paths to replacement components.
  
  ```js
  starlight({
    components: {
      Header: './src/components/CustomHeader.astro',
      Footer: './src/components/CustomFooter.astro',
    }
  })
  ```
</ParamField>

## Advanced Options

<ParamField path="disable404Route" type="boolean" default="false">
  Disable Starlight's default 404 page.
</ParamField>

<ParamField path="prerender" type="boolean" default="true">
  Define whether Starlight pages should be prerendered. Defaults to always prerender Starlight pages, even when the project is set to "server" output mode.
</ParamField>

<ParamField path="markdown" type="object" default="{}">
  Configure features that impact Starlight's Markdown processing.
  
  <ParamField path="markdown.headingLinks" type="boolean" default="true">
    Define whether headings in content should be rendered with clickable anchor links.
  </ParamField>
  
  <ParamField path="markdown.processedDirs" type="array" default="[]">
    Define additional directories where files should be processed by Starlight's Markdown pipeline. Supports local directories relative to the root of your project (e.g., './src/data/comments/').
  </ParamField>
</ParamField>

<ParamField path="routeMiddleware" type="string | array" default="[]">
  Add middleware to process Starlight's route data for each page. Provide a file path or array of file paths to middleware modules.
  
  See the [route data customization guide](https://starlight.astro.build/guides/route-data/) for more information.
</ParamField>
